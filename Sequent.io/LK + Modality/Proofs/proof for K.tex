\RightLabel{\scriptsize implyL}

\RightLabel{\scriptsize exchangeL}
\UnaryInfC{$(C\mathcal{R} G) \rightarrow (G\Vdash A),\forall D.((C\mathcal{R} D) \rightarrow (D\Vdash A \rightarrow B)) \vdash (C\mathcal{R} G) \rightarrow (G\Vdash B)$}

\RightLabel{\scriptsize forallL}

\RightLabel{\scriptsize forallR}

\RightLabel{\scriptsize mBoxL}

\RightLabel{\scriptsize implyR}
\UnaryInfC{$\forall D.((C\mathcal{R} D) \rightarrow (D\Vdash A \rightarrow B)) \vdash (C\Vdash \Box A) \rightarrow (C\Vdash \Box B)$}

\RightLabel{\scriptsize mImplyR}
\UnaryInfC{$\forall D.((C\mathcal{R} D) \rightarrow (D\Vdash A \rightarrow B)) \vdash C\Vdash (\Box A) \rightarrow (\Box B)$}
